# Valid tests
assert add_integer(3, 5) == 8
assert add_integer(100) == 198
assert add_integer(-2, 10) == 8
assert add_integer(0, 0) == 0
assert add_integer(5.5, 2.3) == 7
assert add_integer(10.9, 2) == 12
assert add_integer(3.1) == 101

# Invalid tests (should raise TypeError)
try:
    add_integer("hello", 5)
except TypeError as e:
    assert str(e) == "a must be an integer"

try:
    add_integer(5, "world")
except TypeError as e:
    assert str(e) == "b must be an integer"

try:
    add_integer(None, 5)
except TypeError as e:
    assert str(e) == "a must be an integer"

try:
    add_integer(5, None)
except TypeError as e:
    assert str(e) == "b must be an integer"

try:
    add_integer([1, 2], 5)
except TypeError as e:
    assert str(e) == "a must be an integer"

try:
    add_integer(5, {1: 2})
except TypeError as e:
    assert str(e) == "b must be an integer"
